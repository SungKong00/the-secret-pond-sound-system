# Secret Pond DSSSP Graph EQ Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the current WEQ8C-based inline Graph EQ editor with a DSSSP-based React island while preserving Secret Pond's existing Graph EQ model/render core and Live/Stable audio application contracts.

**Architecture:** Treat DSSSP as the browser graph interaction shell, not as the audio engine or the canonical EQ math. Secret Pond's existing Graph EQ point model, response helpers, Live 1 second debounce/latest-wins render-swap path, and Stable Apply and Restart path remain authoritative. The React island receives normalized layer EQ points from `app.js`, emits immediate draft updates on drag, emits `ended=true` on drag completion, and leaves audio application to the existing settings/live pathways.

**Tech Stack:** Python/FastAPI static dashboard, vanilla `app.js` state owner, React island bundled by esbuild, `dsssp@0.6.4`, `react`/`react-dom`, Playwright, pytest, ruff.

```javascript
export const graphEqDisplayConfig = Object.freeze({
  minFreq: 20,
  maxFreq: 20000,
  minGain: -15,
  maxGain: 15,
  sampleRate: 48000,
});

export const supportedDssspTypes = Object.freeze(["LOWSHELF2", "PEAK", "HIGHSHELF2"]);

export const secretPondToDssspType = Object.freeze({
  low_shelf: "LOWSHELF2",
  bell: "PEAK",
  high_shelf: "HIGHSHELF2",
});

export const dssspToSecretPondType = Object.freeze({
  LOWSHELF1: "low_shelf",
  LOWSHELF2: "low_shelf",
  PEAK: "bell",
  HIGHSHELF1: "high_shelf",
  HIGHSHELF2: "high_shelf",
});

const clamp = (value, min, max) => Math.min(max, Math.max(min, Number(value)));

export const isLockedEndpointPoint = (point) => (
  (point?.type === "low_shelf" && ["low", "legacy-low"].includes(point?.id)) ||
  (point?.type === "high_shelf" && ["high", "legacy-high"].includes(point?.id))
);

export const displayFrequencyForPoint = (point, config = graphEqDisplayConfig) => {
  if (point?.type === "low_shelf" && ["low", "legacy-low"].includes(point?.id)) return config.minFreq;
  if (point?.type === "high_shelf" && ["high", "legacy-high"].includes(point?.id)) return config.maxFreq;
  return clamp(point?.frequency_hz ?? 1000, config.minFreq, config.maxFreq);
};

export const frequencyToX = (frequencyHz, config = graphEqDisplayConfig) => {
  const minLog = Math.log10(config.minFreq);
  const maxLog = Math.log10(config.maxFreq);
  return (Math.log10(clamp(frequencyHz, config.minFreq, config.maxFreq)) - minLog) / (maxLog - minLog);
};

export const gainToY = (gainDb, config = graphEqDisplayConfig) => (
  (config.maxGain - clamp(gainDb, config.minGain, config.maxGain)) / (config.maxGain - config.minGain)
);

export const displayPositionForPoint = (point, config = graphEqDisplayConfig) => ({
  x: frequencyToX(displayFrequencyForPoint(point, config), config),
  y: gainToY(point?.gain_db ?? 0, config),
});

export const toDssspFilters = (points = [], config = graphEqDisplayConfig) => points.map((point) => ({
  id: String(point.id),
  type: secretPondToDssspType[point.type] || "PEAK",
  freq: displayFrequencyForPoint(point, config),
  gain: clamp(point.gain_db ?? 0, config.minGain, config.maxGain),
  q: clamp(point.q ?? 1, 0.1, 18),
}));

export const toSecretPondPoints = (filters = [], previousPoints = []) => filters
  .map((filter, index) => {
    const previous = previousPoints[index] || {};
    const locked = isLockedEndpointPoint(previous);
    return {
      id: String(previous.id || filter.id || `point-${index + 1}`),
      type: dssspToSecretPondType[filter.type] || previous.type || "bell",
      frequency_hz: locked ? Number(previous.frequency_hz) : Math.round(Number(filter.freq || previous.frequency_hz || 1000)),
      gain_db: Number(clamp(filter.gain ?? previous.gain_db ?? 0, graphEqDisplayConfig.minGain, graphEqDisplayConfig.maxGain).toFixed(1)),
      q: Number(clamp(filter.q ?? previous.q ?? 1, 0.1, 18).toFixed(2)),
    };
  })
  .filter((point) => Object.values(dssspToSecretPondType).includes(point.type));

export const fromDssspChangeEvent = (event = {}) => ({
  index: Number(event.index),
  ended: Boolean(event.ended),
  point: {
    type: dssspToSecretPondType[event.type] || "bell",
    frequency_hz: Math.round(Number(event.freq || 1000)),
    gain_db: Number(clamp(event.gain ?? 0, graphEqDisplayConfig.minGain, graphEqDisplayConfig.maxGain).toFixed(1)),
    q: Number(clamp(event.q ?? 1, 0.1, 18).toFixed(2)),
  },
});
```

```jsx
import React, { useEffect, useMemo, useState } from "react";
import { createRoot } from "react-dom/client";
import {
  CompositeCurve,
  FilterPoint,
  FrequencyResponseGraph,
  PointerTracker,
} from "dsssp";
import {
  graphEqDisplayConfig,
  isLockedEndpointPoint,
  toDssspFilters,
  toSecretPondPoints,
} from "./graph_eq_dsssp_adapter.mjs";

const mountedEditors = new WeakMap();

const theme = {
  background: {
    grid: {
      dotted: false,
      lineColor: "rgba(255,255,255,0.12)",
      lineWidth: { minor: 0.5, major: 1, center: 1, border: 1 },
    },
    gradient: { start: "#171b1d", stop: "#080a0b", direction: "vertical" },
    label: { color: "rgba(235,238,232,0.62)", fontSize: 10, fontFamily: "Inter, system-ui, sans-serif" },
    tracker: {
      lineWidth: 1,
      lineColor: "rgba(228,189,91,0.5)",
      labelColor: "#e8d18a",
      backgroundColor: "#111412",
    },
  },
  curve: { width: 2, color: "#e2b84f", opacity: 1 },
  filters: {
    fill: false,
    gradientOpacity: 0.18,
    defaultColor: "#e2b84f",
    colors: [
      { point: "#81c995", curve: "#81c995", active: "#ffffff", drag: "#ffffff", background: "#1f3a2a" },
      { point: "#e2b84f", curve: "#e2b84f", active: "#ffffff", drag: "#ffffff", background: "#3c3218" },
      { point: "#8ab4f8", curve: "#8ab4f8", active: "#ffffff", drag: "#ffffff", background: "#1d2f4d" },
    ],
    point: {
      radius: 13,
      lineWidth: 2,
      backgroundOpacity: { normal: 0.85, active: 1, drag: 1 },
      label: { color: "#f2f1ea", fontFamily: "Inter, system-ui, sans-serif", fontSize: 11 },
    },
    curve: { width: { normal: 1, active: 2 }, opacity: { normal: 0.35, active: 0.75 } },
    zeroPoint: { color: "rgba(255,255,255,0.35)", background: "#101312" },
  },
};

function normalizePoints(points) {
  return Array.isArray(points) ? points : [];
}

function GraphEqDssspEditor({ layerId, points, selectedPointId, disabled, onChange, onSelect, onDragState }) {
  const [localPoints, setLocalPoints] = useState(normalizePoints(points));
  const [dragging, setDragging] = useState(false);
  const filters = useMemo(() => toDssspFilters(localPoints), [localPoints]);
  const selectedIndex = Math.max(0, localPoints.findIndex((point) => point.id === selectedPointId));

  useEffect(() => {
    if (!dragging) setLocalPoints(normalizePoints(points));
  }, [points, dragging]);

  const handleChange = (event) => {
    if (disabled) return;
    const previous = localPoints;
    const nextFilters = [...filters];
    nextFilters[event.index] = {
      ...nextFilters[event.index],
      type: event.type,
      freq: event.freq,
      gain: event.gain,
      q: event.q,
    };
    const nextPoints = toSecretPondPoints(nextFilters, previous);
    const nextPoint = nextPoints[event.index];
    setLocalPoints(nextPoints);
    if (nextPoint) onSelect?.(nextPoint.id);
    onChange?.({ layerId, points: nextPoints, selectedPointId: nextPoint?.id || null, ended: Boolean(event.ended) });
  };

  const handleDrag = (active) => {
    setDragging(active);
    onDragState?.({ layerId, dragging: active });
  };

  return (
    <div className="graph-eq-dsssp-surface" data-graph-eq-dsssp-surface="true">
      <FrequencyResponseGraph
        width={900}
        height={360}
        scale={graphEqDisplayConfig}
        theme={theme}
        style={{ width: "100%", height: "100%" }}
      >
        <CompositeCurve filters={filters} />
        {filters.map((filter, index) => {
          const point = localPoints[index];
          return (
            <FilterPoint
              key={point?.id || index}
              filter={filter}
              index={index}
              active={index === selectedIndex}
              dragX={!isLockedEndpointPoint(point)}
              dragY={!disabled}
              wheelQ={!disabled}
              label={String(index + 1)}
              onChange={handleChange}
              onDrag={handleDrag}
              onEnter={() => point && onSelect?.(point.id)}
            />
          );
        })}
        {!dragging && <PointerTracker />}
      </FrequencyResponseGraph>
    </div>
  );
}

const mountEditor = (container, props) => {
  if (!container) return null;
  let mounted = mountedEditors.get(container);
  if (!mounted) {
    mounted = { root: createRoot(container), props: {} };
    mountedEditors.set(container, mounted);
  }
  mounted.props = { ...mounted.props, ...props };
  mounted.root.render(<GraphEqDssspEditor {...mounted.props} />);
  return mounted;
};

const syncEditor = (container, props) => mountEditor(container, props);

const unmountEditor = (container) => {
  const mounted = mountedEditors.get(container);
  if (!mounted) return;
  mounted.root.unmount();
  mountedEditors.delete(container);
};

window.secretPondGraphEq = Object.freeze({
  mountEditor,
  syncEditor,
  unmountEditor,
  toDssspFilters,
  toSecretPondPoints,
});
```
